  322.     function processAdminUI()

  323.     {

  324.         if ($this->isOwner(User :: getCurrentUser()) || User :: getCurrentUser()->isSuperAdmin())

    333. 

  333.         }
    334. 

  334.  
    335. 

  335.                 $file_mode = $_REQUEST["file_mode".$this->getId()];
    336. 

  336.                 if ($file_mode == "by_upload")
    337. 

  337.                 {
    338. 

  338.                     if(isset($_REQUEST["file_mime_type".$this->getId()]))
    339. 

  339.                         $this->setMimeType($_REQUEST["file_mime_type".$this->getId()]);
    340. 

  340.                     $this->setBinaryDataFromPostVar("file_file_upload".$this->getId());
    341. 

  341.                     $this->setURL(null);
    342. 

  342.                     // Reset the remote file size ( not used )
    343. 

  343.                     $this->setRemoteFileSize(0);
    344. 

  344.                 }
    345. 

  345.                 else
    346. 

  346.                 {
    347. 

  347.                     if ($file_mode == "remote")
    348. 

  348.                     {
    349. 

  349.                         $this->setURL($_REQUEST["file_url".$this->getId()]);
    350. 

  350.                         $this->setBinaryDataOid(null);
    351. 

  351.                         // When switching from local to remote, this field does not exist yet
    352. 

  352.                         if (isset($_REQUEST["file_old_remote_size".$this->getId()]))
    353. 

  353.                         {
    354. 

  354.                             if ($_REQUEST["file_remote_size".$this->getId()] != $_REQUEST["file_old_remote_size".$this->getId()])
    355. 

  355.                                 $this->setRemoteFileSize($_REQUEST["file_remote_size".$this->getId()]);
    356. 

  356.                         }
    357. 

  357.                         else
    358. 

  358.                             $this->setRemoteFileSize(0);
    359. 

  359.                     }
    360. 

  360.                 }
    361. 

  361.             }
    362. 

  362.         }
    363. 

  363.     }
